{{ $prompt }}plakar at ~/backups create
 
{{ $prompt }}plakar at ~/backups backup /etc

{{ $prompt }}plakar at ~/backups ls
003affd1 9.8 MB 0s /etc
{{ $prompt }}export PLAKAR_PASSPHRASE=-HelloWorld-  # Keep your passphrase safe
{{ $prompt }}plakar at sftp://nestor@nas.plakar.io/var/backups create
 
{{ $prompt }}plakar at ~/backups sync to sftp://nestor@nas.plakar.io/var/backups
sync: synchronized 1 snapshot
info: sync: synchronization completed: 1 snapshots synchronized
 
{{ $prompt }}plakar at sftp://nestor@nas.plakar.io/var/backups ls
2025-04-08T10:14:37Z 003affd1 9.8 MB 0s /etc
{{ $prompt }}export PLAKAR_PASSPHRASE=-HelloWorld-  # Keep your passphrase safe
{{ $prompt }}
{{ $prompt }}plakar at ~/backups check --fast # Structural check
check: verification of 003affd1:/etc completed successfully
 
{{ $prompt }}plakar at ~/backups check # data integrity check
check: verification of 003affd1:/etc completed successfully
{{ $prompt }}# Everything is good our data is safe
{{ $prompt }}export PLAKAR_PASSPHRASE=-HelloWorld-  # Keep your passphrase safe
{{ $prompt }}# Let's start the UI to search and restore only one file
{{ $prompt }}plakar at ~/backups ui
 
{{ $prompt }}# Or we can fully restore a specific backup
{{ $prompt }}plakar at ~/backups ls # Let's find the right backup
003affd1 9.8 MB 0s /etc
{{ $prompt }}plakar at ~/backups restore -to /tmp/backups 003affd1
